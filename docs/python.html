Array Operations

import numpy as np

# Create arrays
a = np.array([1, 2, 3])

c = np.zeros((3, 4))

# Broadcasting
x = np.array([[1], [2], [3]])

z = x + y  # shape (3,4)

# Linear algebra
A = np.random.rand(100, 100)
eigvals, eigvecs = np.linalg.eig(A)

Plotting

import matplotlib.pyplot as plt

x = np.linspace(0, 2*np.pi, 100)
plt.plot(x, np.sin(x), label='sin(x)')
plt.plot(x, np.cos(x), label='cos(x)')
plt.legend()
plt.xlabel('x')
plt.ylabel('y')
plt.show()
